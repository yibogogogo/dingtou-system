### 4.2 估值评分公式

```python
# 估值评分（满分25分，红利类）
def valuation_score(pe, pb, dividend_yield, pe_percentile):
    score = 0
    
    # PE评分（满分8分）
    if pe < 6: score += 8
    elif pe < 7: score += 6
    elif pe < 8: score += 4
    elif pe < 9: score += 2
    else: score += 0
    
    # PB评分（满分7分）
    if pb < 0.7: score += 7
    elif pb < 0.8: score += 5
    elif pb < 0.9: score += 3
    elif pb < 1.0: score += 1
    else: score += 0
    
    # 股息率评分（满分5分）
    if dividend_yield > 5.5: score += 5
    elif dividend_yield > 5.0: score += 4
    elif dividend_yield > 4.5: score += 3
    elif dividend_yield > 4.0: score += 2
    else: score += 1
    
    # 历史分位评分（满分5分）
    if pe_percentile < 10: score += 5
    elif pe_percentile < 20: score += 4
    elif pe_percentile < 30: score += 3
    elif pe_percentile < 50: score += 2
    else: score += 0
    
    return score
```

## 八、风险控制与退出机制

### 8.1 止损规则

```
【单标止损】
- 定投后标的下跌超过20%：触发评估，检查基本面是否恶化
- 定投后标的下跌超过30%：触发止损，暂停该标的定投
- 连续3个月评分<25：暂停该标的，转入观察

【组合止损】
- 整体组合浮亏超过15%：全面评估策略有效性
- 整体组合浮亏超过25%：考虑减仓至50%，保留现金
- 整体组合浮亏超过35%：启动"极端行情"应对预案
```

### 8.2 止盈规则

```
【动态止盈】
- 单个标的盈利超过30%：考虑减仓1/3
- 单个标的盈利超过50%：考虑减仓1/2
- 综合评分进入F级（0-24分）：考虑全部止盈

【目标止盈】
- 年化收益率达到15%：可部分止盈（减仓20%）
- 年化收益率达到25%：建议大幅止盈（减仓50%）
- 年化收益率达到35%：建议全部止盈，等待下次机会

【再投资规则】
- 止盈后的资金不立即投入，等待评分≥B级（55分）再投入
- 可转入货币基金或短债基金暂存
```

### 8.3 极端行情应对

```
【极端下跌（如2022年）】
- 触发条件：连续2个月评分<25
- 应对策略：
  1. 暂停所有定投
  2. 将原计划定投资金转入货币基金
  3. 等待评分回升至B级（55分）以上
  4. 一次性投入累积资金

【极端上涨（如2021年初）】
- 触发条件：连续2个月评分>80
- 应对策略：
  1. 暂停定投
  2. 逐步减仓至50%
  3. 等待评分回落至C级（40-54分）再恢复
```

## 十一、Python 回测代码框架

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from datetime import datetime, timedelta

class MultiFactorDingTou:
    """
    三标的多因子量化择时定投策略

    """
    
    def __init__(self, initial_capital=2000):

        """计算RSI"""
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    def calculate_kdj(self, high, low, close, n=9, m1=3, m2=3):
        """计算KDJ"""
        lowest_low = low.rolling(window=n).min()
        highest_high = high.rolling(window=n).max()
        rsv = (close - lowest_low) / (highest_high - lowest_low) * 100
        
        k = pd.Series(index=close.index)
        d = pd.Series(index=close.index)
        
        k.iloc[:n] = 50
        d.iloc[:n] = 50
        
        for i in range(n, len(close)):
            k.iloc[i] = (2/3) * k.iloc[i-1] + (1/3) * rsv.iloc[i]
            d.iloc[i] = (2/3) * d.iloc[i-1] + (1/3) * k.iloc[i]
        
        j = 3 * d - 2 * k
        return k, d, j
    
    def calculate_bias(self, prices, period):
        """计算BIAS"""
        ma = prices.rolling(window=period).mean()
        bias = (prices - ma) / ma * 100
        return bias
    
    def calculate_technical_score(self, prices, high, low, close):
        """计算技术指标评分（满分40分）"""
        # RSI评分
        rsi6 = self.calculate_rsi(close, 6)
        rsi12 = self.calculate_rsi(close, 12)
        rsi24 = self.calculate_rsi(close, 24)
        
        rsi_score = 0
        if rsi6.iloc[-1] < 20: rsi_score += 15
        elif rsi6.iloc[-1] < 30: rsi_score += 10
        elif rsi6.iloc[-1] < 40: rsi_score += 5
        
        if rsi12.iloc[-1] < 25: rsi_score += 10
        elif rsi12.iloc[-1] < 35: rsi_score += 5
        
        if rsi24.iloc[-1] < 30: rsi_score += 10
        elif rsi24.iloc[-1] < 40: rsi_score += 5
        
        # KDJ评分
        k, d, j = self.calculate_kdj(high, low, close)
        kdj_score = 0
        if k.iloc[-1] < 20: kdj_score += 10
        if d.iloc[-1] < 30: kdj_score += 10
        if j.iloc[-1] < 10: kdj_score += 10
        if k.iloc[-1] > d.iloc[-1] and k.iloc[-2] <= d.iloc[-2]:  # 金叉
            kdj_score += 10
        
        # BIAS评分
        bias5 = self.calculate_bias(close, 5)
        bias10 = self.calculate_bias(close, 10)
        bias20 = self.calculate_bias(close, 20)
        bias60 = self.calculate_bias(close, 60)
        
        bias_score = 0
        if bias5.iloc[-1] < -5: bias_score += 5
        if bias10.iloc[-1] < -8: bias_score += 5
        if bias20.iloc[-1] < -10: bias_score += 10
        if bias60.iloc[-1] < -15: bias_score += 10
        
        total_score = min(rsi_score + kdj_score + bias_score, 40)
        return total_score
    
    def calculate_valuation_score(self, pe, pb, dividend_yield, pe_percentile):
        """计算估值评分（满分25分）"""
        score = 0
        
        # PE评分
        if pe < 6: score += 8
        elif pe < 7: score += 6
        elif pe < 8: score += 4
        elif pe < 9: score += 2
        
        # PB评分
        if pb < 0.7: score += 7
        elif pb < 0.8: score += 5
        elif pb < 0.9: score += 3
        elif pb < 1.0: score += 1
        
        # 股息率评分
        if dividend_yield > 5.5: score += 5
        elif dividend_yield > 5.0: score += 4
        elif dividend_yield > 4.5: score += 3
        elif dividend_yield > 4.0: score += 2
        else: score += 1
        
        # 历史分位评分
        if pe_percentile < 10: score += 5
        elif pe_percentile < 20: score += 4
        elif pe_percentile < 30: score += 3
        elif pe_percentile < 50: score += 2
        
        return min(score, 25)
    
    def calculate_momentum_score(self, prices):
        """计算动量评分（满分15分）"""
        score = 0
        
        # 月度动量
        monthly_return = (prices.iloc[-1] - prices.iloc[-20]) / prices.iloc[-20] * 100
        if monthly_return < -10: score += 5
        elif monthly_return < -5: score += 3
        elif monthly_return < 0: score += 1
        
        # 季度动量
        quarterly_return = (prices.iloc[-1] - prices.iloc[-60]) / prices.iloc[-60] * 100
        if quarterly_return < -20: score += 5
        elif quarterly_return < -10: score += 3
        elif quarterly_return < 0: score += 1
        
        # 年度动量
        if len(prices) >= 250:
            yearly_return = (prices.iloc[-1] - prices.iloc[-250]) / prices.iloc[-250] * 100
            if yearly_return < -30: score += 5
            elif yearly_return < -15: score += 3
            elif yearly_return < 0: score += 1
        
        return min(score, 15)
    
    def calculate_total_score(self, prices, high, low, close, 
                             pe=None, pb=None, dividend_yield=None, 
                             pe_percentile=None):
        """计算综合评分"""
        technical = self.calculate_technical_score(prices, high, low, close)
        
        if pe is not None:
            valuation = self.calculate_valuation_score(pe, pb, dividend_yield, pe_percentile)
        else:
            valuation = 12.5  # 默认值
        
        momentum = self.calculate_momentum_score(prices)
        
        # 市场情绪和资金流向评分（简化处理）
        sentiment = 5  # 默认中性
        fund_flow = 5  # 默认中性
        
        total = technical + valuation + momentum + sentiment + fund_flow
        return min(total, 100)
    
    def run_backtest(self, data_dict):
        """
        执行回测
        data_dict: {标的名称: DataFrame}
        """
        # 获取共同的交易日
        common_dates = data_dict['科创50'].index
        for name in data_dict:
            common_dates = common_dates.intersection(data_dict[name].index)
        
        # 按月执行定投
        monthly_dates = pd.date_range(start=common_dates[0], end=common_dates[-1], freq='MS')
        
        for date in monthly_dates:
            # 找到最近的交易日
            valid_dates = [d for d in common_dates if d >= date]
            if not valid_dates:
                continue
            trade_date = valid_dates[0]
            
            scores = {}
            for name, df in data_dict.items():
                if trade_date not in df.index:
                    continue
                
                # 获取历史数据
                hist_data = df.loc[:trade_date]
                if len(hist_data) < 250:
                    continue
                
                # 计算评分
                score = self.calculate_total_score(
                    hist_data['close'],
                    hist_data['high'],
                    hist_data['low'],
                    hist_data['close']
                )
                scores[name] = score
            
            # 根据评分分配资金
            if scores:
                total_score = sum(scores.values())
                for name, score in scores.items():
                    if score >= 40:  # C级以上
                        amount = self.initial_capital * (score / total_score)
                        # 执行买入（简化处理）
                        self.positions[name] += amount / data_dict[name].loc[trade_date, 'close']
                        self.trade_log.append({
                            'date': trade_date,
                            'name': name,
                            'amount': amount,
                            'score': score
                        })
        
        return self.trade_log
```
